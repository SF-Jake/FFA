	desc = "EVTDESC70000"

		country = {

						}
					}
				}

				factor = 0

						province_id = 94 #Nevada
						province_id = 97 #Utah
						province_id = 117 #N. Dakota
						province_id = 120 #S. Dakota
						province_id = 81 #Oregon
						province_id = 78 #Washington
						province_id = 159 #Wisconsin
						province_id = 110 #Wyoming
						is_core = CAN
					}
				}
			}
			modifier = {
				factor = 100
				owner = { tag = USA }

					any_neighbor_province = {

					is_slave = no

country_event = {
    id = 70004
    title = "EVTNAME70004"
    desc = "EVTDESC70004"
    #picture = "Western"
    trigger = {
		OR = {
			any_owned_province = {
				has_pop_type = labourers
				is_core = PRO
				NOT = { has_province_modifier = mine_expansion }
			}
			any_owned_province = {
				has_pop_type = farmers
				is_core = PRO
				NOT = { has_province_modifier = agriculture_expansion }
			}
			any_owned_province = {
				OR = {
					province_id = 1481
					province_id = 1496
					province_id = 1498
					province_id = 1538
					province_id = 1566
					province_id = 1569
					province_id = 2615
					province_id = 2659
					province_id = 1499
					province_id = 1496
					province_id = 1608
				}
				is_core = PRO
				has_province_flag = treaty_port
				NOT = { has_province_modifier = treaty_port }
			}
		}
    }
    is_triggered_only = yes
    allow_multiple_instances = yes

    option = {
        name = "EVTOPTA70004"
		any_owned = {
			limit = {
				OR = {
					province_id = 1481
					province_id = 1496
					province_id = 1498
					province_id = 1538
					province_id = 1566
					province_id = 1569
					province_id = 2615
					province_id = 2659
					province_id = 1499
					province_id = 1496
					province_id = 1608
				}
				is_core = PRO
				has_province_flag = treaty_port
				NOT = { has_province_modifier = treaty_port }
			}
			add_province_modifier = { name = treaty_port duration = -1 }
		}
		any_owned = {
			limit = {
				has_pop_type = labourers
				is_core = PRO
				NOT = { has_province_modifier = mine_expansion }
				NOT = { has_province_modifier = treaty_port }
			}
			add_province_modifier = { name = mine_expansion duration = -1 }
		}
		any_owned = {
			limit = {
				has_pop_type = farmers
				is_core = PRO
				NOT = { has_province_modifier = agriculture_expansion }
			}
			add_province_modifier = { name = agriculture_expansion duration = -1 }
		}
    }
}
